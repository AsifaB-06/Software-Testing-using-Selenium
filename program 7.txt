//Test the case to submit a login form using web driver

package ex7sublogin;  
import org.openqa.selenium.By;  
import org.openqa.selenium.Keys;  
import org.openqa.selenium.WebDriver;  
import org.openqa.selenium.chrome.ChromeDriver;  
public class login  
{  
 public static void main(String[] args)  
	{  
		System.setProperty("webdriver.chrome.driver","/home/skasc/Downloads/chromedriver"); 
		WebDriver driver = new ChromeDriver();  
		driver.get("https://www.google.com");  
		driver.findElement(By.linkText("Sign in")).click(); 
		driver.findElement(By.id("identifierId")).sendKeys("abc@gmail.com"); 
		driver.findElement(By.id("identifierId")).sendKeys(Keys.ESCAPE);  
 	}  
}
